\begin{frame}[t]{Lógica de Programação com Django}
	\fontsize{14pt}{15.2}\selectfont{
		Tutorial de boas-vindas com Django.  \\
		
		
	}\par
	\vspace{1em}
	
	\fontsize{10pt}{15}\selectfont{
		\begin{itemize}%[<+->]  
			\item instalação: faça a instalação do django. Comece repetindo o comando abaixo no terminal e se funcionar já era.... 
			
			\begin{beamercolorbox}[wd=\textwidth]{warning}
				|> py -m pip install Django
			\end{beamercolorbox}
			
			\item criação do projeto: entre no diretório que pretende salvar o projeto e repita o comando abaixo no terminal.
			
			\begin{beamercolorbox}[wd=\textwidth]{warning}
				|> django-admin startproject django\_project .
			\end{beamercolorbox}
			
			\item execução: entre no diretório do projeto que foi criado e repita o comando abaixo no terminal.
			
			\begin{beamercolorbox}[wd=\textwidth]{warning}
				|> python manage.py runserver
			\end{beamercolorbox}
			
			\item criação de app: repita o comando abaixo no terminal.
			
			\begin{beamercolorbox}[wd=\textwidth]{warning}
				|> python manage.py startapp pages
			\end{beamercolorbox}
			
			
		\end{itemize}
	}\par
	\vspace{1em}
	
\end{frame}
